```
class Vehicle:
    def __init__(self, typ, speed, price):
        self.type = typ
        self.speed = speed
        self.price = price

    def drive(self):
        print("%s正在行驶中~~" % self.type)

    def __str__(self):
        return "当前正在运行的交通工具是：%s，速率：%dkm/h，价格：%d元" %(self.type, self.speed, self.price)


plane = Vehicle("飞机", 400, 1000)
train = Vehicle("火车", 200, 400)
ship = Vehicle("轮船", 180, 300)

plane.drive()
train.drive()
ship.drive()

print(plane)
print(train)
print(ship)
```
